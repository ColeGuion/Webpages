x = "This happened over a Decade ago." y = "This happened three DECADES ago." re.search(r"decade", x, re.IGNORECASE) # <re.Match object; span=(21, 27), match='Decade'> re.search(r"decade", y, re.I) # <re.Match object; span=(20, 26), match='DECADE'> Check if the word "decade" is in the string.
x = "This happened over a Decade ago." y = "This happened three DECADES ago." re.search(r"\bdecade\b", x, re.I) # <re.Match object; span=(21, 27), match='Decade'> re.search(r"\bdecade\b", y, re.I) # None Check if the EXACT word "decade" is in the string.
text = "Cats and dogs are friends." pattern = r"\b(cat|dog)s?\b" re.findall(pattern, text, re.I) # ['Cat', 'dog'] Match Multiple Options
text = "Order #123 cost $45 and took 7 days." re.findall(r"\d+", text) # ['123', '45', '7'] Extract All Numbers

text = "abc123" match = re.search(r"abc(?=\d{3})", text) print(bool(match)) # True, 'abc' is followed by 3 digits Positive Lookahead: Match this position only if it is followed by this pattern.
